def calculator(number):
    if number == 1:
        print("Enter your first number")
        word1 = input()
        print("Enter your second number")
        word2 = input()
        word3 = int(word1) + int(word2)
        print(word1 + " + " + word2 + " = " + str(word3))
    if number == 2: 
        print("Enter your first number")
        word1 = input()
        print("Enter your second number")
        word2 = input()
        word3 = int(word1) - int(word2)
        print(word1 + " - " + word2 + " = " + str(word3))
    if number == 3:
        print("Enter your first number")
        word1 = input()
        print("Enter your second number")
        word2 = input()
        word3 = int(word1) / int(word2)
        print(word1 + " / " + word2 + " = " + str(word3))
    if number == 4:
        print("Enter your first number")
        word1 = input()
        print("Enter your second number")
        word2 = input()
        word3 = int(word1) * int(word2)
        print(word1 + " * " + word2 + " = " + str(word3))

calculator(1)
calculator(2)
calculator(3)
calculator(4)
